Motivation

Die Korrelation zwischen Social-Media Aktivitt und Aktienpreisen wurde bereits in einer vielzahl von Forschungsarbeiten untersucht, wobei wiederholt die Schlussfolgerung erreicht wird, dass Nennungen alleine kein zuverlssiger Indikator fr signifikante positive Preisbewegungen sind. Grund dafr sind die beidseitige Wechselwirkung zwischen Social-Media Aktivitt und Asset-Preis sowie das Vermischen positiver und negativer Sentimente in den Nennungen. Vereinfacht gesagt, die Social-Media Daten beinhalten zu viele Strsignale und mssen zunchst aufwendig gefiltert werden um jeglichen Nutzen fr die Preisvorhersage zu liefern.

Diese Exploration wurde Motiviert durch die Hypothese, dass die Aktivitt in sozialen Netzwerken bezglich Assets von geringer popularitt bedeutend weniger Strsignale enthlt und daher, potenziell bereits in roher Form, einen besseren Indikator fr Preisbewegungen liefert.

symbol <- "DOGE"

sample_df <- as_tibble(read.csv(file=sprintf("data/price_histories/%s.csv", symbol))) %>%
    mutate(time=as.Date(as.POSIXct(time, origin="1970-01-01"))) %>%
    filter(open > 0 | close > 0 | high > 0 | low > 0)

sample_df %>%
    ggplot(aes(x=time, y=close)) +
    geom_candlestick(aes(open=open, high=high, low=low, close=close)) +
    geom_ma(ma_fun = SMA, n = 50, color = "black", size = 1) +
    geom_smooth(se=FALSE) +
    labs(title=symbol, x="Time", y="Price (USD)")

# indicator
sma <- function(x, n=50) {
    stats::filter(x, rep(1/n, n), sides = 1)
}

relative_delta <- function(x, n=50) {
    len = length(x)
    ma <- sma(x,n)
    relative_delta = list()
    for (i in 1:len) {
        relative_delta[i] <- if (is.na(ma[i])) NA else (x[i] - ma[i])/ma[i]
    }
    return(unlist(relative_delta))
}

ggplot(data.frame(x=1:nrow(sample_df), y=relative_delta(sample_df$close))) +
    geom_area(mapping=aes(x=x, y=y, fill=(y > 0))) +
    geom_line(aes(x=x, y=y)) +
    geom_hline(yintercept=2, linetype='dotted') +
    geom_hline(yintercept=1, linetype='dotted') +
    geom_hline(yintercept=-1, linetype='dotted') +
    geom_hline(yintercept=-2, linetype='dotted') +
    guides(fill="none") +
    labs(title="Relative Delta", x="Time", y="y")
